def getTotalX(a, b):
    # Write your code here
    
    factors = list(range(a[0], b[0]+1)) # List of possible factors
    
    for i in a:
        factors = list(filter(lambda x: x % i == 0, factors))
    
    for j in b:
        factors = list(filter(lambda x: j % x == 0, factors))
    # print(factors)
    
    return len(factors)
