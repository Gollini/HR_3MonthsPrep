def plusMinus(arr):
    # Write your code here
    n = len(arr)
    ratios = [0]*3 # (pos, neg, zero)
    
    for i in arr:
        if i > 0: 
            ratios[0] += 1
        elif i < 0:
            ratios[1] += 1
        elif i == 0:
            ratios[2] += 1
            
    for r in ratios:
        print(round(r/n, 6))
        
    

if __name__ == '__main__':
    n = int(input().strip())

    arr = list(map(int, input().rstrip().split()))

    plusMinus(arr)
